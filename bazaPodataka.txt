--Kreiranje baze

CREATE DATABASE OnlineShop
USE OnlineShop


--Kreiranje svih tabela


CREATE TABLE VrsteArtikla(
    IDVrsteArtikla int PRIMARY KEY IDENTITY (1,1) NOT NULL,
    vrsta_artikla char(20) UNIQUE NOT NULL
);

CREATE TABLE Boje(
    IDBoje int PRIMARY KEY IDENTITY (1,1) NOT NULL,
    naziv_boje CHAR(15) UNIQUE NOT NULL,
    sifra CHAR(20) UNIQUE NOT NULL
);

CREATE TABLE Nabavljaci(
    IDNabavljaca int PRIMARY KEY IDENTITY (1,1) NOT NULL,
    naziv_nabavljaca CHAR(30) NOT NULL,
    broj NVARCHAR(20) NOT NULL
);

CREATE TABLE Velicine(
    IDVelicine int PRIMARY KEY IDENTITY (1,1) NOT NULL,
    Velicina NVARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE Artikli(
    IDArtikla int PRIMARY KEY IDENTITY (1,1) NOT NULL,
    IDVrsteArtikla int NOT NULL REFERENCES VrsteArtikla(IDVrsteArtikla),
    IDNabavljaca int NOT NULL REFERENCES Nabavljaci(IDNabavljaca),
    IDBoje int NOT NULL REFERENCES Boje(IDBoje),
    spol VARCHAR(10) NOT NULL,
    cijena VARCHAR(10) NOT NULL,
    dodatne_informacije VARCHAR(100)
);

CREATE TABLE Slike(
    IDSlike int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    IDArtikla int NOT NULL REFERENCES Artikli(IDArtikla),
    url_slike NVARCHAR(100) NOT NULL
);

CREATE TABLE Stanje(
    IDStanja int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    IDArtikla int NOT NULL REFERENCES Artikli(IDArtikla),
    IDVelicine int NOT NULL REFERENCES Velicine(IDVelicine),
    Stanje int NOT NULL
);

CREATE TABLE Korisnik(
IDKorisnika int PRIMARY KEY IDENTITY(1,1) NOT NULL,
ime NVARCHAR(200) NOT NULL,
prezime NVARCHAR(200) NOT NULL,
adresa_stanovanja NVARCHAR(250) NOT NULL,
datum_rodjenja DATE NOT NULL,
kontakt_telefon NVARCHAR(100) NOT NULL,
);

CREATE TABLE Narudzba(
    IDNarudzbe int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    IDKorisnika int NOT NULL REFERENCES Korisnik(IDKorisnika),
    cijena VARCHAR(10) NOT NULL,
    vrijeme DATETIME NOT NULL DEFAULT (getdate())
)

CREATE TABLE Korpa(
    IDKorpe int PRIMARY KEY IDENTITY(1,1) NOT NULL,
    IDNarudzbe int NOT NULL REFERENCES Narudzba(IDNarudzbe),
    IDArtikla int NOT NULL REFERENCES Artikli(IDArtikla),
    komada int NOT NULL,
    cijena VARCHAR(10) NOT NULL,
)
